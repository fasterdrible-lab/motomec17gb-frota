import requests
import logging

logger = logging.getLogger(__name__)

class FIPEApi:
    def __init__(self):
        """Inicializa conexao com FIPE API"""
        try:
            self.base_url = 'https://parallelum.com.br/fipe/api/v1'
            self.session = requests.Session()
            logger.info("FIPE API inicializada")
        except Exception as e:
            logger.error(f"Erro FIPE API: {e}")
            self.base_url = None

    def test_connection(self):
        """Testa conexao"""
        try:
            response = self.session.get(f"{self.base_url}/carros/marcas", timeout=5)
            return response.status_code == 200
        except:
            return False
